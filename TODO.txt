CREATE TABLE public.Application (

  created_at timestamp with time zone NOT NULL DEFAULT now(),

  CONSTRAINT Application_campaign_task_id_fkey FOREIGN KEY (campaign_task_id) REFERENCES public.Campaign_Task(id),
  CONSTRAINT Application_volunteer_id_fkey FOREIGN KEY (volunteer_id) REFERENCES public.Volunteer(id)
);
CREATE TABLE public.Barangay (
  id bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  name character varying,
  malnourished_count bigint,
  severity smallint DEFAULT '0'::smallint,
  risk_score double precision,
  last_updated_date date,
  CONSTRAINT Barangay_pkey PRIMARY KEY (id)
);
CREATE TABLE public.Barangay_Data (
  id bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  date_collected date,
  barangay_id bigint,
  malnourished_count bigint,
  CONSTRAINT Barangay_Data_pkey PRIMARY KEY (id),
  CONSTRAINT Barangay_Data_barangay_id_fkey FOREIGN KEY (barangay_id) REFERENCES public.Barangay(id)
);
CREATE TABLE public.Campaign (
  id bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  title character varying,
  description character varying,
  organization_id bigint,
  CONSTRAINT Campaign_pkey PRIMARY KEY (id),
  CONSTRAINT Campaign_organization_id_fkey FOREIGN KEY (organization_id) REFERENCES public.Organization(id)
);
CREATE TABLE public.Campaign_Task (
  id bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  campaign_id bigint,
  task_name character varying,
  task_description character varying,
  start_date timestamp without time zone,
  volunteer_quota bigint,
  CONSTRAINT Campaign_Task_pkey PRIMARY KEY (id)
);
CREATE TABLE public.Organization (
  id bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  created_at timestamp with time zone NOT NULL,
  name character varying,
  type smallint DEFAULT '0'::smallint,
  contact_person character varying,
  email character varying,
  contact_person_info character varying DEFAULT ''::character varying,
  address character varying,
  CONSTRAINT Organization_pkey PRIMARY KEY (id)
);
CREATE TABLE public.Volunteer (
  id bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  f_name character varying,
  email character varying,
  l_name character varying,
  CONSTRAINT Volunteer_pkey PRIMARY KEY (id)
